// ================================================ // 법인카드 관리 시스템 - Google Apps Script 서버 // ================================================ // 설치 위치: Google Sheets > 확장 프로그램 > Apps Script const SHEET_NAME = 'records'; // ------------------------------------------------ // 공통: 시트 가져오기 // ------------------------------------------------ function getSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow([ 'id','name','lab','card','cert', 'rent','returnTime','useDate','amount', 'merchant','project','step','matched','createdAt' ]); } return sheet; } // ------------------------------------------------ // GET 요청 처리 (데이터 조회 + payload 방식 쓰기) // ------------------------------------------------ function doGet(e) { const action = e.parameter.action; const payload = e.parameter.payload; try { // payload가 있으면 쓰기 요청 (CORS 우회) if (payload) { const data = JSON.parse(payload); if (data.action === 'step1') return jsonResponse(saveStep1Raw(data), e); if (data.action === 'step2') return jsonResponse(saveStep2Raw(data), e); if (data.action === 'match') return jsonResponse(saveMatchRaw(data), e); } if (action === 'search') { return jsonResponse(searchRaw(e.parameter.name, e.parameter.card), e); } if (action === 'all') { return jsonResponse(getAllRecordsRaw(), e); } return jsonResponse({ error: '알 수 없는 action' }, e); } catch(err) { return jsonResponse({ error: err.message }, e); } } // ------------------------------------------------ // POST 요청 처리 (데이터 저장/수정) // ------------------------------------------------ function doPost(e) { const data = JSON.parse(e.postData.contents); const action = data.action; try { if (action === 'step1') return saveStep1(data); if (action === 'step2') return saveStep2(data); if (action === 'match') return saveMatch(data); return jsonResponse({ error: '알 수 없는 action' }); } catch(err) { return jsonResponse({ error: err.message }); } } // ------------------------------------------------ // 1단계: 대여 신청 저장 // ------------------------------------------------ function saveStep1Raw(data) { const sheet = getSheet(); const id = Utilities.getUuid(); const createdAt = new Date().toISOString(); sheet.appendRow([ id, data.name, data.lab, data.card, data.cert, data.rent, '', // returnTime '', // useDate '', // amount '', // merchant '', // project 1, // step false, // matched createdAt ]); return { success: true, id: id }; } function saveStep1(data) { return jsonResponse(saveStep1Raw(data)); } // ------------------------------------------------ // 2단계: 반납 & 지출 신고 저장 // ------------------------------------------------ function saveStep2Raw(data) { const sheet = getSheet(); const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0] === data.id) { sheet.getRange(i + 1, 7).setValue(data.returnTime); sheet.getRange(i + 1, 8).setValue(data.useDate); sheet.getRange(i + 1, 9).setValue(data.amount); sheet.getRange(i + 1, 10).setValue(data.merchant); sheet.getRange(i + 1, 11).setValue(data.project); sheet.getRange(i + 1, 12).setValue(2); return { success: true }; } } return { error: '해당 대여 신청서를 찾을 수 없습니다.' }; } function saveStep2(data) { return jsonResponse(saveStep2Raw(data)); } // ------------------------------------------------ // 이름 + 카드번호로 1단계 완료 건 검색 // ------------------------------------------------ function searchRaw(name, card) { const sheet = getSheet(); const rows = sheet.getDataRange().getValues(); const results = []; for (let i = 1; i < rows.length; i++) { const r = rows[i]; if (r[1] === name && String(r[3]) === String(card) && Number(r[11]) === 1) { results.push(rowToObj(r)); } } return { results: results }; } function search(name, card) { return jsonResponse(searchRaw(name, card)); } // ------------------------------------------------ // 전체 레코드 조회 (관리자용) // ------------------------------------------------ function getAllRecordsRaw() { const sheet = getSheet(); const rows = sheet.getDataRange().getValues(); const records = []; for (let i = 1; i < rows.length; i++) { records.push(rowToObj(rows[i])); } return { records: records }; } function getAllRecords() { return jsonResponse(getAllRecordsRaw()); } // ------------------------------------------------ // 대조 결과 matched 상태 업데이트 // ------------------------------------------------ function saveMatchRaw(data) { const sheet = getSheet(); const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { sheet.getRange(i + 1, 13).setValue(false); } data.matchedIds.forEach(function(id) { for (let i = 1; i < rows.length; i++) { if (rows[i][0] === id) { sheet.getRange(i + 1, 13).setValue(true); break; } } }); return { success: true }; } function saveMatch(data) { return jsonResponse(saveMatchRaw(data)); } // ------------------------------------------------ // 헬퍼: 행 배열 → 객체 변환 // ------------------------------------------------ function rowToObj(r) { return { id: r[0], name: r[1], lab: r[2], card: r[3], cert: r[4], rent: r[5], returnTime: r[6], useDate: r[7], amount: r[8], merchant: r[9], project: r[10], step: Number(r[11]), matched: r[12] === true || r[12] === 'TRUE', createdAt: r[13] }; } // ------------------------------------------------ // 헬퍼: JSON 응답 생성 (JSONP 지원) // ------------------------------------------------ function jsonResponse(obj, e) { const callback = e && e.parameter && e.parameter.callback; if (callback) { return ContentService .createTextOutput(callback + '(' + JSON.stringify(obj) + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
